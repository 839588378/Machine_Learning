import numpy as np

def performance_metric2(y_true, y_predict):
    """"""
    y_true_arr = np.arr(y_true)
    y_predict_arr = np.arr(y_predict)
    
    #
    y_true_mean = np.mean(y_true_arr)
    
    ss_tot=0
    ss_reg=0
    ss_res=0
    
    for index in range(y_true_arr):
        ss_tot += (y_true_arr[index] - y_true_mean)**2
    
    for index in range(y_true_arr):
        ss_reg += (y_predict_arr[index] - y_true_mean)**2
        
    for index in range(y_true_arr):
        ss_res += (y_predict_arr[index] - y_true_arr[index])**2
    score = 1 - (ss_res/ss_tot)

    return score

问题 5 - 回答:¶

当最大深度为1的时候，R2评分低，模型有较大的偏差，两条线间距小，方差小；当最大深度为10时候，R2评分高偏差较小，两条线距离加大，方差大。¶

当模型能很好匹配训练集和测试集的时候，偏差小，模型适应性好；当模型在训练集上表现好，而在测试集表现不好，表明拟合度过高复杂度高，方差大¶

当最大深度为1训练时，模型的预测是出现了很大的偏差，因为模型不够复杂不足以捕捉基本关系，其score不足0.6，得分太低正是模型复杂度太低所致；当模型以最大深度10训练时，出现了很大的方差，即出现了过拟合现象，这是因为模型过于复杂，对于训练集而言可以获得很好的匹配，但是无法将此模型泛化到更多数据，测试集得分太低，解决此问题的方法一是增大数据量二是降低模型复杂度。¶

问题 8 - 回答：¶

1、在K折交叉验证中，我们用到的数据是训练集中的所有数据。我们将训练集的所有数据平均划分成K份（通常选择K=10），取第K份作为验证集，它的作用就像我们用来估计高考分数的模拟题，余下的K-1份作为交叉验证的训练集。¶

5 max_depth=1 2-101 max_depth=1 13-102 max_depth=1 1-910 max_depth=1 10101010 max_depth=1

2、对于一组参数，首先调节其中一个影响最大的参数，直到调节为最优，然后再调节第二个参数到最优。从而选择一组最佳参数组合。而其通过调节其cv参数来实现与较差验证来实现最佳参数的选择。¶

3、cvresults返回一个dict，包含相关参数以及对应的值。比如'param_kernel','param_gamma','param_degree'.¶

4、可以在一定程度上减小过拟合,这个方法的优势在于,同时重复运用随机产生的子样本进行训练和验证,每次的结果验证一次.这个方法用于训练的数据只比整体数据集少了一个样本，因此最接近原始样本的分布。¶

def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    n=len(X)/cv
    
    X1,y1=X[:n],y[:n]
    X2,y2=X[n:2*n],y[n:2*n]
    X3,y3=X[2*n:],y[2*n:]
    
#     
    estimator1,estimator2,estimator3=estimator,estimator,estimator
    0
#   
    estimator1.fit(np.append(X2,X3, axis =0),np.append(y2,y3,axis =0))
    estimator2.fit(np.append(X1,X3, axis =0),np.append(y1,y3,axis =0))
    estimator3.fit(np.append(X1,X2, axis =0),np.append(y1,y2,axis =0))

#     X,XY
    y1_predict=estimator1.predict(X1)
    y2_predict=estimator1.predict(X2)
    y3_predict=estimator1.predict(X3)
    
#     R2
    scores = [performance_metric(y1,y1_predict),performance_metric(y2,y2_predict),performance_metric(y3,y3_predict)]
    return scores

def OverwirtePredictTrials(X, y, fitter, data):
    """ Performs trials of fitting and predicting data. """

    # Store the predicted prices
    prices = []

    for k in range(10):
        # Split the data
        X_train, X_test, y_train, y_test = train_test_split(X, y, 
            test_size = 0.2, random_state = k)
        
        # Fit the data
        reg = fitter(X_train, y_train)
        
        # Make a prediction
        pred = reg.predict([data])[0]
        prices.append(pred)
        
        # Result
        print "Trial {}: {:,.2f} ".format(k+1, pred)

    # Display price range
    print "\nRange in prices: {:,.2f} ".format(max(prices) - min(prices))
